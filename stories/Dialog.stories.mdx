<Preview>
  <Story name="Dialog">
    {() => {
      const [opened, setOpened] = useState(true)
      return (
        <div className="bg-gray-300" style={{ height: '100vh' }}>
          <Button
            className="bg-blue-400 p-2 text-white border-none"
            onClick={() => { setOpened((value) => { return !value })}}
          >toggle dialog</Button>
          <Dialog
            open={opened}
            focusTrap={true}
            onDismiss={() => {
              setOpened(false)
            }}
          >
            <div className="w-64 h-40"></div>
            <button
              className="bg-red-400 p-2 text-white"
              onClick={() => { setOpened((value) => { return !value })}}
            >close</button>
          </Dialog>
        </div>
      )
    }}
  </Story>
