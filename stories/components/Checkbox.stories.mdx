# 多选框 Checkbox

在一组选择项中进行多选。

### 简单使用

<Canvas>
  <Story name="Checkbox">
    {() => {
      const [flag, setFlag] = useState(false)
      return (
        <Checkbox
          label={'Apple'}
          value={flag}
          onChange={(value) => { console.log(value); setFlag(value) }}
        />
      )
    }}
  </Story>
</Canvas>

### 不可用状态

<Canvas>
  <Story name="Disabled Checkbox">
    {() => {
      const [flag, setFlag] = useState(false)
      return (
        <div>
          <Checkbox
            className="mr-2"
            label={'Apple'}
            checked={flag}
            onChange={(value) => { console.log(value); setFlag(value) }}
            disabled
          />
          <Checkbox
            className="mr-2"
            label={'Banana'}
            checked={!flag}
            onChange={(value) => { console.log(value); setFlag(value) }}
            disabled
          />
        </div>
      )
    }}
  </Story>
</Canvas>

### 一组多选框

<Canvas>
  <Story name="Group of Checkbox">
    {() => {
      const [n, setN] = useState(["apple"])
      return (
        <CheckboxGroup value={n} onChange={(value) => { setN(value) }}>
          <Checkbox label={'Apple'} value={'apple'}></Checkbox>
          <Checkbox label={'Banana'} value={'banana'}></Checkbox>
          <Checkbox label={'Cherry'} value={'cherry'}></Checkbox>
        </CheckboxGroup>
      )
    }}
  </Story>
</Canvas>
