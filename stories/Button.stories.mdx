# 按钮 Button

### 加载状态

<Canvas>
  <Story name="Loading Button">
    {() => {
      const [loading, setLoading] = useState(true)
      const [loading2, setLoading2] = useState(false)
      return (
        <div>
          <div>
            <Button
              className="mr-2 mb-2"
              onClick={() => { console.log('clicked!') }}
              loading>
                loading
            </Button>
            <Button
              className="mr-2 mb-2"
              styling={{ type: 'primary' }}
              onClick={() => { console.log('clicked!') }}
              loading>
                loading
            </Button>
            <Button
              className="mr-2 mb-2"
              styling={{ type: 'text' }}
              onClick={() => { console.log('clicked!') }}
              loading>
                loading
            </Button>
          </div>
          <div>
            <Button
              className="mr-2 mb-2"
              onClick={() => { console.log('clicked!') }}
              loading={loading}>
                {loading? null :'loading'}
            </Button>
            <Button
              className="mr-2 mb-2"
              onClick={() => {
                setLoading2(true)
                setTimeout(() => {
                  setLoading2(false)
                }, 3000)
              }}
              loading={loading2}>
                click to loading
            </Button>
          </div>
        </div>
      )
    }}
  </Story>
</Canvas>
