        { key: 'name', title: "Name" },
        { key: 'age', title: "Age" },
        { key: 'gender', title: "Gender" },
        { key: 'mobile', title: "Mobile" },
        { key: 'email', title: "Email" },
        { key: 'profession', title: "Profession" },

            i.email,
            i.profession,
          ],
        }
      })
      return (
        <div>
          <Switch value={loading} onChange={(value) => switchLoading(value)} />
          <LoadingCover loading={loading} label={'loading...'}>
            <Table
              columns={columns}
              rows={rows}
            />
          </LoadingCover>
        </div>
      )
    }}
